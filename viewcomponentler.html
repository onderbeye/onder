using Microsoft.AspNetCore.Mvc;
using System.Threading.Tasks;
using Books.Model;
using System.Linq;


namespace Books.ViewComponents
{
public class YazarlarViewComponent : ViewComponent

{
private readonly kitap_dbContext db = new kitap_dbContext();
public YazarlarViewComponent(kitap_dbContext _DbContext)
{
this.db = _DbContext;
}

public async Task InvokeAsync()
{
var Yazarlar= (from x in db.Yazarlars
select new YazarBadgeVM
{
Id= x.Id,
Adi= x.Adi,
Soyadi=x.Soyadi,
KitapSayisi = (from k in db.Kitaplars
where k.YazarId == x.Id
select k).Count() }).OrderByDescending(i=>i.KitapSayisi).Take(10).ToList();
return View(Yazarlar);

}
}

public class YazarBadgeVM
{
public int Id{ get; set; }
public string Adi{ get; set; }
public string Soyadi{ get; set; }
public int KitapSayisi{ get; set; }


}

}
